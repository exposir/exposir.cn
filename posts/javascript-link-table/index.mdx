class Node {
  constructor(val) {
    this.val = val;
    this.next = null;
  }
}

class LinkedList {
  constructor(val) {
    this.head = new Node(val);
  }

  findNodeByVal(val) {
    let curr = this.head;
    while (curr != null && curr.val != val) {
      curr = curr.next;
    }
    return curr ? curr : -1;
  }

    let curr = this.findNodeByVal(val);
